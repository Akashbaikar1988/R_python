import pandas as pd
import statsmodels.api as sm
from statsmodels.formula.api import ols
from statsmodels.iolib.summary2 import summary_col

# Load data from R using reticulate
data1=pd.DataFrame(r["data1"])
# Two variable regression
ols_model1 = ols("mpg ~ wt", data=data1).fit()

# Multi variable regression
ols_model2 = ols("mpg ~ wt + disp + hp", data=data1).fit()

# Factor variable as independent variable
ols_model3 = ols("mpg ~ wt + C(gear)", data=data1).fit()

# Change the base category of a factor variable
ols_model4 = ols("mpg ~ wt + C(gear, Treatment(reference=4))", data=data1).fit()

# Interaction between one category and one continuous variable
ols_model5 = ols("mpg ~ wt + C(gear) + wt:C(gear)", data=data1).fit()

# Combine regression result table
results_table = summary_col(
    results=[ols_model1, ols_model2, ols_model3, ols_model4, ols_model5],
    model_names=["Model-1", "Model-2", "Model-3", "Model-4", "Model-5"],
    stars=True,
    info_dict={
        "N. Obs": lambda x: f"{int(x.nobs)}",
        "Adj R2": lambda x: f"{x.rsquared_adj:.3f}"
    }
)

print(results_table)
